% Application with profile details and attachments / Bewerbung mit Profildaten und Anlagen
\newcommand{\ModeApplication}{%
  \UseFoldmarkstrue
  \UseFootertrue
  \UseFooterColorBlocktrue
  \UseHeaderColorBlocktrue
  \UseBusinessLegalFooterfalse
  \UsePageTotaltrue
  \UseContinuationHeadertrue
  \UseGradientPaneltrue
  \UseLogofalse
  \UseIconsfalse
  % Compact Form A header: name plus one contact row.
  \UsePositionfalse
  \UseLinkedInfalse
  \UseWebsitefalse
  \UseAttachmentstrue
  \UseReferenceBlockfalse
  \renewcommand{\AttachmentPreset}{application}%
}

%---------------------------------------------------------------
%               Mode selection / Modusauswahl
%---------------------------------------------------------------
% Falls back to application mode / Fällt auf den Bewerbungsmodus zurück
\newcommand{\ApplyDocumentMode}{%
  \ifdefstring{\DocumentMode}{application}{\ModeApplication}{%
    \ifdefstring{\DocumentMode}{authority}{\ModeAuthority}{%
      \ifdefstring{\DocumentMode}{legal}{\ModeLegal}{%
        \ifdefstring{\DocumentMode}{business}{\ModeBusiness}{%
          \ifdefstring{\DocumentMode}{minimal}{\ModeMinimal}{\ModeApplication}%
        }%
      }%
    }%
  }%
}

% Apply default or selected mode / Standard- oder gewählten Modus anwenden
\ApplyDocumentMode
