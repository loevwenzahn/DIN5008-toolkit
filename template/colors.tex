%---------------------------------------------------------------
%                 Color mode / Farbmodus
%---------------------------------------------------------------
% Maps the selected mode to active document colors / Ordnet dem gewählten Modus die aktiven Dokumentfarben zu
\newcommand{\ApplyColorMode}{%
  \ifdefstring{\ColorMode}{blue}{%
    \colorlet{accent}{accentblue}
    \colorlet{panelone}{panelblueone}
    \colorlet{paneltwo}{panelbluetwo}
    \colorlet{panelthree}{panelbluethree}
    \colorlet{panelfour}{panelbluefour}
    \colorlet{noticebackground}{panelblueone}
    \colorlet{linkurlcolor}{accent}
    \colorlet{linktextcolor}{accent}
  }{%
    \ifdefstring{\ColorMode}{gray}{%
      \colorlet{accent}{accentgray}
      \colorlet{panelone}{panelgrayone}
      \colorlet{paneltwo}{panelgraytwo}
      \colorlet{panelthree}{panelgraythree}
      \colorlet{panelfour}{panelgrayfour}
      \colorlet{noticebackground}{panelgrayone}
      \colorlet{linkurlcolor}{accent}
      \colorlet{linktextcolor}{accent}
    }{%
      \ifdefstring{\ColorMode}{black}{%
        \colorlet{accent}{neutralblack}
        \colorlet{panelone}{neutralwhite}
        \colorlet{paneltwo}{neutralwhite}
        \colorlet{panelthree}{neutralwhite}
        \colorlet{panelfour}{neutralwhite}
        \colorlet{noticebackground}{neutralwhite}
        \colorlet{linkurlcolor}{neutralblack}
        \colorlet{linktextcolor}{neutralblack}
      }{%
      \ifdefstring{\ColorMode}{print}{%
        \colorlet{accent}{printaccent}
        \colorlet{panelone}{panelprintone}
        \colorlet{paneltwo}{panelprinttwo}
        \colorlet{panelthree}{panelprintthree}
        \colorlet{panelfour}{panelprintfour}
        \colorlet{noticebackground}{panelprintone}
        \colorlet{linkurlcolor}{textdark}
        \colorlet{linktextcolor}{textdark}
      }{%
        \colorlet{accent}{accentgreen}
        \colorlet{panelone}{panelgreenone}
        \colorlet{paneltwo}{panelgreentwo}
        \colorlet{panelthree}{panelgreenthree}
        \colorlet{panelfour}{panelgreenfour}
        \colorlet{noticebackground}{panelgreenone}
        \colorlet{linkurlcolor}{accent}
        \colorlet{linktextcolor}{accent}
      }%
      }%
    }%
  }%
  \hypersetup{%
    urlcolor=linkurlcolor,
    linkcolor=linktextcolor,
    citecolor=linktextcolor,
    filecolor=linktextcolor
  }%
}

% Apply default or selected mode / Standard- oder gewählten Modus anwenden
\ApplyColorMode

%---------------------------------------------------------------
%          Public hexadecimal color overrides / Hex-Farben
%---------------------------------------------------------------
% Use exactly six hexadecimal characters without a leading #. Apply custom
% colors after \UsePreset so the preset establishes the remaining roles first.
% Genau sechs Hexadezimalzeichen ohne führendes # verwenden. Eigene Farben
% nach \UsePreset setzen, damit die Vorgabe zuerst alle übrigen Rollen setzt.
\ExplSyntaxOn
\cs_new_protected:Npn \din_toolkit_set_html_color:nn #1#2
  {
    \regex_match:nnTF { \A [0-9A-Fa-f]{6} \Z } {#2}
      { \definecolor{#1}{HTML}{#2} }
      {
        \PackageError{letter-template}{Invalid~hex~color~`#2'}
          {In~content.tex,~use~exactly~six~hexadecimal~characters~without~a~leading~hash.}
      }
  }

\cs_new_protected:Npn \din_toolkit_refresh_link_colors:
  {
    \hypersetup
      {
        urlcolor=linkurlcolor,
        linkcolor=linktextcolor,
        citecolor=linktextcolor,
        filecolor=linktextcolor
      }
  }
